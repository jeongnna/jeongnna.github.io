```python
def auroc(y_true, y_score):
    sorted_idx = np.argsort(-y_score)
    y_true = y_true[sorted_idx]
    
    inc = 0
    auc = 0
    for y in y_true:
        if y == 1:
            inc += 1
        else:
            auc += inc
            
    n = len(y_true)
    auc /= inc * (n - inc)
    
    return auc
```
